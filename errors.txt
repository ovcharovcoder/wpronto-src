# Перевірте, що всі папки існують
Test-Path "C:\WPLaunch\core"
Test-Path "C:\WPLaunch\config"
Test-Path "C:\WPLaunch\data"
Test-Path "C:\WPLaunch\template"
Test-Path "C:\WPLaunch\www"

# Перевірте, що Inno Setup встановлено
Test-Path "C:\Program Files (x86)\Inno Setup 6\ISCC.exe"

--------------------------------------------------------------


# Крок 0: Зупинити всі процеси (ВАЖЛИВО!)
Write-Host "Зупиняємо сервер..." -ForegroundColor Yellow
taskkill /F /IM nginx.exe 2>$null
taskkill /F /IM php-cgi.exe 2>$null
taskkill /F /IM mysqld.exe 2>$null
Start-Sleep -Seconds 2

# Крок 1: Перехід у папку проєкту (назва папки залишилась WPLaunchGUI)
cd "C:\WPronto\src\WPLaunchGUI"

# Крок 2: Компіляція
dotnet build -c Release

# Крок 3: Публікація (створює WProntoGUI.exe завдяки налаштуванням проекту)
dotnet publish -c Release -o "C:\WPronto\publish"

# Крок 4: Копіювання папок з сервером (ДОДАНО "logs")
$folders = @("core", "config", "data", "template", "www", "logs")
foreach ($folder in $folders) {
    if (Test-Path "C:\WPronto\$folder") {
        Copy-Item -Path "C:\WPronto\$folder" -Destination "C:\WPronto\publish\" -Recurse -Force
        Write-Host "✅ Скопійовано: $folder"
    }
}

# Крок 5: Копіювання текстових файлів
Copy-Item -Path "C:\WPronto\about.txt" -Destination "C:\WPronto\publish\" -Force -ErrorAction SilentlyContinue
Copy-Item -Path "C:\WPronto\license.txt" -Destination "C:\WPronto\publish\" -Force -ErrorAction SilentlyContinue
Write-Host "✅ Текстові файли скопійовано" -ForegroundColor Green

# Крок 6: Створення інсталятора (опціонально)
if (Test-Path "C:\Program Files (x86)\Inno Setup 6\ISCC.exe") {
    if (Test-Path "C:\WPronto\installer_final.iss") {
        & "C:\Program Files (x86)\Inno Setup 6\ISCC.exe" "C:\WPronto\installer_final.iss"
        Write-Host "✅ Інсталятор створено" -ForegroundColor Green
    } else {
        Write-Host "⚠ Файл installer_final.iss не знайдено" -ForegroundColor Yellow
    }
} else {
    Write-Host "⚠ Inno Setup не знайдено, пропускаємо створення інсталятора" -ForegroundColor Yellow
}

# Показуємо розмір збірки
$publishSize = [math]::Round((Get-ChildItem "C:\WPronto\publish" -Recurse -ErrorAction SilentlyContinue | Measure-Object -Property Length -Sum).Sum / 1MB, 2)
Write-Host "`n📦 Розмір збірки: $publishSize MB" -ForegroundColor Cyan

Write-Host "`n✅ Готово! Збірка в C:\WPronto\publish" -ForegroundColor Green
Write-Host "🚀 Запустіть: C:\WPronto\publish\WProntoGUI.exe" -ForegroundColor Cyan
